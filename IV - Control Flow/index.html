IV - Control Flow

Note: Open browser console before demo.

For...in

const person = {
name: "Mosh",
age: 30
};
for (let key in person)
console.log(key, person[key]);


const colors = ["red", "green", "blue"];
for (let index in colors)
console.log(index, colors[index]);

For...of

const colors = ["red", "green", "blue"];
for (let color of colors)
console.log(color);

Break and Continue

Example:
let i = 0;
while (i <= 10) {
console.log(i);
i++;
}

... with break:
let i = 0;
while (i <= 10) {
if (i === 5) break;
console.log(i);
i++;
}

... with continue:
let i = 0;
while (i <= 10) {
if (i % 2 === 0) {
i++;
continue;
}
console.log(i);
i++;
}
